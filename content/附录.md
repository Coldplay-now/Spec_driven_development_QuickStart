type UserRole = 'admin' | 'user' | 'guest';
type HttpMethod = 'GET' | 'POST' | 'PUT' | 'DELETE';

interface ApiConfig {
  method: HttpMethod;
  url: string;
  headers?: Record<string, string>;
  data?: unknown;
}

// 4. 工具类型
type CreateUserRequest = Omit<User, 'id' | 'createdAt'>;
type UpdateUserRequest = Partial<Pick<User, 'name' | 'email' | 'active'>>;
type UserSummary = Pick<User, 'id' | 'name' | 'email'>;

// 5. 类和装饰器
class UserService {
  private apiClient: ApiClient;
  
  constructor(apiClient: ApiClient) {
    this.apiClient = apiClient;
  }
  
  async getUser(id: number): Promise<User> {
    const response = await this.apiClient.get<ApiResponse<User>>(
      `/users/${id}`
    );
    return response.data;
  }
  
  async createUser(userData: CreateUserRequest): Promise<User> {
    const response = await this.apiClient.post<ApiResponse<User>>(
      '/users',
      userData
    );
    return response.data;
  }
  
  async updateUser(
    id: number, 
    updates: UpdateUserRequest
  ): Promise<User> {
    const response = await this.apiClient.put<ApiResponse<User>>(
      `/users/${id}`,
      updates
    );
    return response.data;
  }
}
